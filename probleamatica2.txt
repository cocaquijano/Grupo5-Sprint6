CREATE VIEW IF NOT EXISTS vista1_clientes AS 
SELECT cliente.customer_id as ID, cliente.branch_id as NumeroSucursal, cliente.customer_name as Nombre, cliente.customer_surname as Apellido, cliente.customer_DNI as DNI, cast(strftime('%Y.%m%d', 'now')-strftime('%Y.%m%d', dob) as INTEGER) as Edad
FROM cliente
WHERE Edad >= 40
ORDER BY customer_DNI DESC


CREATE VIEW IF NOT EXISTS vista2_clientes AS 
SELECT cliente.customer_id as ID, cliente.branch_id as NumeroSucursal, cliente.customer_name as Nombre, cliente.customer_surname as Apellido, cliente.customer_DNI as DNI, cast(strftime('%Y.%m%d', 'now')-strftime('%Y.%m%d', dob) as INTEGER) as Edad
FROM cliente
WHERE cliente.customer_name = "Anne" or cliente.customer_name = "Tyler"
ORDER BY Edad ASC
